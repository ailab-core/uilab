## Usage
```tsx
import {
  DataList,
  DataListItem,
  DataListLabel,
  DataListValue,
} from '@/components/data-list';
```

```tsx
<DataList>
  <DataListItem>
    <DataListLabel>Name</DataListLabel>
    <DataListValue>John Doe</DataListValue>
  </DataListItem>
  <DataListItem>
    <DataListLabel>Email</DataListLabel>
    <DataListValue>john.doe@example.com</DataListValue>
  </DataListItem>
</DataList>
```
