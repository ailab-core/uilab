## Usage
```tsx
import {
  Collapsible,
  CollapsibleContent,
  CollapsibleTrigger,
} from "uilab-core"
```
```tsx
<Collapsible>
  <CollapsibleTrigger>Can I use this in my project?</CollapsibleTrigger>
  <CollapsibleContent>
    Yes. Free to use for personal and commercial projects. No attribution
    required.
  </CollapsibleContent>
</Collapsible>
```
