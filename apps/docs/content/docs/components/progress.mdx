### Animated Progress

<ComponentPreview title="Animated Progress">
  <Progress value={75} className="transition-all duration-300" />
</ComponentPreview>

```tsx
import { Progress } from 'uilab-core'
import { useEffect, useState } from 'react'

export function Example() {
  const [progress, setProgress] = useState(0)

  useEffect(() => {
    const timer = setInterval(() => {
      setProgress((prev) => (prev >= 100 ? 0 : prev + 10))
    }, 500)
    return () => clearInterval(timer)
  }, [])

  return <Progress value={progress} className="transition-all duration-300" />
}
```

## Props

- `value`: `number` - The progress value (0-100)
- `max`: `number` - The maximum progress value (default: 100)
