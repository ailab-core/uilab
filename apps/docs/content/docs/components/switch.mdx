### Controlled

<ComponentPreview title="Controlled Switch">
  <div className="flex items-center gap-2">
    <Switch id="controlled" checked={true} onCheckedChange={(checked) => console.log(checked)} />
    <Label htmlFor="controlled">Controlled switch</Label>
  </div>
</ComponentPreview>

```tsx
import { Switch, Label } from 'uilab-core'
import { useState } from 'react'

export function Example() {
  const [enabled, setEnabled] = useState(false)

  return (
    <div className="flex items-center gap-2">
      <Switch
        id="controlled"
        checked={enabled}
        onCheckedChange={setEnabled}
      />
      <Label htmlFor="controlled">
        {enabled ? 'Enabled' : 'Disabled'}
      </Label>
    </div>
  )
}
```

## Props

- `checked`: `boolean` - The controlled checked state
- `defaultChecked`: `boolean` - The default checked state (uncontrolled)
- `onCheckedChange`: `(checked: boolean) => void` - Event handler called when the state changes
- `disabled`: `boolean` - When true, prevents interaction
- `required`: `boolean` - When true, indicates the switch must be checked
- `name`: `string` - The name for form submission
- `value`: `string` - The value for form submission
