## Usage:
Add it into the providers:
```tsx
import { Toaster } from "@/components/blocks/sonner"

function App() {
  <OtherProviders>
    <Toaster />
    <MyApp />
  </OtherProviders>
}
```

```tsx
import { toast } from "sonner"
```
```tsx
toast("Event has been created.")
```
